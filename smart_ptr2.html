在上篇的结尾，我们写出一个toy的初级shared_ptr。智能指针不仅仅自动回收资源，而且要和裸指针表现一致，本文将致力于完善它，最终将会呈现给大家一个可以完全替代裸指针的智能指针类。

类型转换

假如有一个Base类和继承自它的一个子类Derived，那么存在从Derived指针到Base指针的隐式转换：

   Derived*  pd = new Derived;
   Base* pb = pd; //  隐式转换
};

我们写出类似的智能指针版本:

    bert::shared_ptr<Derived> pd(new Derived);
    bert::shared_ptr<Base>     pb(pd); // compile error

  template<typename Iterator, typename Distance >
    inline void
    advance(Iterator& it, Distance n)     {
      // concept requirements -- taken care of in __advance
      std::__advance(it, n, std::__iterator_category(it));
    }
   //  std::__iterator_category获取迭代器的类型


  template<typename BidirectionalIterator, typename Distance>
    inline void
    __advance(BidirectionalIterator& it, Distance n,
              bidirectional_iterator_tag)    {
      if (n > 0)
        while (n--)
            ++it;   //双向迭代器，O(n)复杂度
      else
        while (n++)
            --it;
    }

  template<typename RandomAccessIterator, typename Distance>
    inline void
    __advance(RandomAccessIterator& it, Distance n,
              random_access_iterator_tag)    {
         it += n; // 随机迭代器，O(1)复杂度
    }

sharedptr是引用计数的指针，因此比起裸指针，需要添加一个引用计数的对象。使用int变量作为引用计数，我们很容易得出它的第一个版本的结构，如图1，执行bert::sharedptr ptr1(new T)之后：

由于较为简单，这里直接放上第一版代码，略去一些暂时无关紧要的函数：

#ifndef BERT_SHAREPTR_H
#define BERT_SHAREPTR_H

namespace bert
{

template <typename T>
class shared_ptr
{
public:
    explicit shared_ptr(T* ptr = 0) : ptr_(ptr)
    {
        count_ = new int(1);
    }

    ~shared_ptr()
    {
        if (-- *count_ == 0)
        {
            delete  count_;
            delete  ptr_;
        }
    }

    shared_ptr(const shared_ptr& other) : count_(other.count_), ptr_(other.ptr_)
    {
        ++ *count_;
    }

    shared_ptr& operator=(const shared_ptr& other)
    {
        if (this == &other)
            return *this;

        reset();

        count_ = other.count_;
        ptr_ = other.ptr_;
        ++ *count_;

        return *this;
    }

    void reset(T* ptr = 0)
    {
        if (ptr == ptr_)  return;
        shared_ptr(ptr).swap(*this);
    }

    void swap(shared_ptr& other)
    {
        if (this != &other)
        {
            // please include header: C++98: <algorithm>, C++11 : <utility>
            std::swap(count_, other.count_);
            std::swap(ptr_, other.ptr_);
        }
    }

    T& operator*() const
    {
        return *ptr_;
    }

    T* operator->() const
    {
        return ptr_;
    }

    T* get() const
    {
        return ptr_;
    }

private:
    int* count_;
    T*   ptr_;
};

} // end namespace bert

#endif
